8   import java.util.ArrayList;
9   import java.util.Random;

11  /**
12   * Luo Kaytavat huoneiden välille ja arpoo niiden aloitusreunan huoneessa.
13   */
14  public class Kaytavat {
15  
16      private int[][] kartta;
17      private ArrayList<Huone> sijoitetutHuoneet;
18      private ArrayList<Huone> sijoitetutHuoneetPoisto;
19      private Huone nykyinenHuone;
20      private Huone seuraavaHuone;
21      private int kaytavienMaara;
22  
23      /**
24       * Annetaan tarvittavat tiedot käytävien luomiseen ja kartta niiden
25       * sijoittamiseen.
26       *
27       * @param kartta Luolaston kartta, jonne käytävät sijoitetaan.
28       * @param sijoitetutHuoneet Sijoitetut huoneet, joiden välille sijoitetaan
29       * käytäviä.
30       */
31      public Kaytavat(int[][] kartta, ArrayList<Huone> sijoitetutHuoneet) {
32          this.kartta = kartta;
33          this.sijoitetutHuoneet = sijoitetutHuoneet;
34          this.kaytavienMaara = 0;
35      }
36  
37      /**
38       * Käy läpi sijoitettujen käytävien listaa ja sijoittaa käytävän nykyisen ja
39       * seuraavan huoneen välille päivittäen niiden arvoja.
40       *
41       * @return Palauttaa luotujen käytävien määrän testejä varten.
42       */
43      public int luoKaytavat() { // Luodaan käytävät huoneiden välillä jotta niissä pystyy liikkumaan
44          sijoitetutHuoneetPoisto = new ArrayList(sijoitetutHuoneet);
45          seuraavaHuone = new Huone(0, 0, 0);
46          if (!sijoitetutHuoneetPoisto.isEmpty()) {
47              nykyinenHuone = sijoitetutHuoneetPoisto.remove(sijoitetutHuoneetPoisto.size() - 1); // Otetaan viimeisin huone
48  //            System.out.println("sijoitetut huoneet poisto size: " + sijoitetutHuoneetPoisto.size());
49              while (!sijoitetutHuoneetPoisto.isEmpty()) { // aloitetaan käymään läpi
50                  if (seuraavaHuone.getHuoneenLeveys() != 0) { // jos ei vielä määritetty seuraavaa huonetta, eli ensimmäinen kerta !!!AIHEUTTI VIAN JOSSA UUDEN KERROKSEN LUONTI EI TOIMI
51                      nykyinenHuone = seuraavaHuone; // muutetaan huone johon luotiin käytävä huoneeksi, josta aloitetaan luomaan käytävä
52                  }
53                  seuraavaHuone = sijoitetutHuoneetPoisto.remove(sijoitetutHuoneetPoisto.size() - 1); // otetaan huone, johon ensimmäisestä luodaan käytävä
54  //                System.out.println("sijoitetut huoneet poisto size: " + sijoitetutHuoneetPoisto.size());
55                  arvoAloitusPaikat();
56              }
57          }
58          return this.kaytavienMaara;
59      }
60  
61      /**
62       * Arpoo neljästä vaihtoehdosta käytävän aloituspaikan. 0 = huoneen yläosa 1
63       * = vasen sivu 2 = oikea sivu 3 = alaosa.
64       */
65      public void arvoAloitusPaikat() { // Arpoo käytävän aloituspaikat
66          Random rng = new Random();
67          int ehto = rng.nextInt(4); // luodaan 4 vaihtoehtoa 0 = huoneen yläosa 1 = vasen sivu 2 = oikea sivu 3 = alaosa
68          kayVaihtoehdotLapi(ehto);
69      }
70  
71      /**
72       * Arvotun vaihtoehdon perusteella vaihtaa aloituspaikan koordinaatteja
73       * niin, että ne ovat oikealla sivustalla ja lähtevät liikkumaan oikeaan
74       * suuntaan.
75       *
76       * @param ehto Huoneen sivu, jolta käytävä aloitetaan 0 = huoneen yläosa 1 =
77       * vasen sivu 2 = oikea sivu 3 = alaosa.
78       */
79      public void kayVaihtoehdotLapi(int ehto) {
80          Random rng = new Random();
81          Huone huone = nykyinenHuone;
82          int huoneenLeveys = huone.getHuoneenLeveys();
83          int muutettavaX = huone.getX() - 1;
84          int muutettavaY = huone.getY() - 1;
85          int aloitusSuunta;
86          if (ehto == 1 || ehto == 2) { // Vasen tai oikea sivu
87              aloitusSuunta = 0; // aloitetaan siirtämällä y
88              muutettavaY += 1 + rng.nextInt(huoneenLeveys);
89          } else { // ylös tai alas
90              aloitusSuunta = 1; // aloitetaan siirtämällä x
91              muutettavaX += 1 + rng.nextInt(huoneenLeveys);
92          }
93          if (ehto == 2) {
94              muutettavaX += huoneenLeveys + 1;
95          } else if (ehto == 3) {
96              muutettavaY += huoneenLeveys + 1;
97          }
98          Huone maaranpaa = this.seuraavaHuone; // otetaan maaranpaaksi seuraava huone
99          int paaX = maaranpaa.getX() + rng.nextInt(maaranpaa.getHuoneenLeveys()); // arvotaan jokin x-koordinaatti johon käytävä liitetään
100         int paaY = maaranpaa.getY() + rng.nextInt(maaranpaa.getHuoneenLeveys()); // arvotaan jokin y-koordinaatti johon käytävä liitetään
101         teeKaytava(muutettavaX, muutettavaY, paaX, paaY, aloitusSuunta);
102     }
103 
104     /**
105      * Luo käytävän kahden pisteen välille, muuttamalla ensin x tai y
106      * koordinaatit oikeaan kohtaan ja sen jälkeen jäljelle jäävän suunnan.
107      *
108      * @param x Käytävän aloituspaikan x-koordinaatti.
109      * @param y Käytävän aloituspaikan y-koordinaatti.
110      * @param pX Käytävän lopetuspaikan x-koordinaatti.
111      * @param pY Käytävän lopetuspaikan y-koordinaatti.
112      * @param aloitusSuunta Suunta, johon käytävää ensin aletaan luomaan, 0 = y
113      * ja 1 = x;
114      */
115     public void teeKaytava(int x, int y, int pX, int pY, int aloitusSuunta) { // Tehdaan kaytava kahden pisteen valille
116         int muutettavaX = x;
117         int muutettavaY = y;
118         int aloitus = aloitusSuunta;
119         kartta[muutettavaY][muutettavaX] = 1;
120         Random rng = new Random();
121         Huone maaranpaa = this.seuraavaHuone; // otetaan maaranpaaksi seuraava huone
122         int paaX = pX;
123         int paaY = pY;
124         while (true) {
125             if (paaX == muutettavaX && paaY == muutettavaY) {
126                 this.kaytavienMaara++;
127                 break;
128             }
129             if (aloitus == 0) {
130                 aloitus = 1;
131                 if (muutettavaY < paaY) { // katsotaan onko lähtö y pienempi vai suurempi, ja liikutaan sen mukaan
132                     while (muutettavaY != paaY) { // jos y pienempi kasvatetaan y kunnes saman arvoinen
133                         muutettavaY++;
134                         kartta[muutettavaY][muutettavaX] = 1;
135                     }
136                 } else {
137                     while (muutettavaY != paaY) { //jos y suurempi vähennetään y kunnes saman arvoinen
138                         muutettavaY--;
139                         kartta[muutettavaY][muutettavaX] = 1;
140                     }
141                 }
142             } else {
143                 aloitus = 0;
144                 if (muutettavaX < paaX) { // sama x:n arvoille y:n jälkeen
145                     while (muutettavaX != paaX) {
146                         muutettavaX++;
147                         kartta[muutettavaY][muutettavaX] = 1;
148                     }
149                 } else {
150                     while (muutettavaX != paaX) {
151                         muutettavaX--;
152                         kartta[muutettavaY][muutettavaX] = 1;
153                     }
154                 }
155             }
156         }
157     }

175     public void setSeuraavaHuone(Huone seur) {

177             this.seuraavaHuone = seur;
178         }
179     }
