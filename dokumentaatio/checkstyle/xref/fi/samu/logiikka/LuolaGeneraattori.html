59      public LuolaGeneraattori(int koko, int huoneenLeveysMinimi, int huoneenLeveysMaksimi) {
60          if (koko < 10) {
61              this.koko = 10;  // Kartan minimikoko tällä hetkellä 25
62          } else {
63              this.koko = koko;  // Kartan sivun pituus, vaatii kokeilua oikean löytämiseksi
64          }
65          if (huoneenLeveysMinimi < 3 || huoneenLeveysMinimi > 8) {
66              this.huoneenLeveysMinimi = 3;  // huoneen minimikoko tällä hetkellä 3
67          } else {
68              this.huoneenLeveysMinimi = huoneenLeveysMinimi;
69          }
70          if (huoneenLeveysMaksimi > 8 || huoneenLeveysMaksimi < 3) {
71              this.huoneenLeveysMaksimi = 8;  // huoneen maksimikoko tällä hetkellä 8
72          } else {
73              this.huoneenLeveysMaksimi = huoneenLeveysMaksimi;
74          }

113     public void alustaKartta(int koko) { // Nollaa kartan kaikki alkiot
114         kartta = new int[koko][koko];
115         for (int i = 0; i < koko; i++) {
116             for (int j = 0; j < koko; j++) {

122     /**
123      * Luo listan huoneista perustuen täyttöasteeseen joka on määritelty
124      * aiemmin.
125      */
126     public void luoHuoneet() { // Luo listan huoneista, jotka sijoitetaan karttaan.
127         Random rng = new Random();
128         huoneLista = new ArrayList();
129         int hlmin = huoneenLeveysMinimi;
130         int hlmax = huoneenLeveysMaksimi;
131         int kaytettyPintaAla = 0;
132         while (kaytettyPintaAla < tayttoAste) {
133             int huoneenLeveys = hlmin + rng.nextInt(1 + hlmax - hlmin); // Huoneiden koon määritys tässä 3-8 leveys ja korkeus.
134             huoneLista.add(huoneenLeveys);
135             kaytettyPintaAla += huoneenLeveys * huoneenLeveys;
136         }
137     }

139     /**
140      * Pyrkii sijoittamaan aiemmin luodut huoneet kartalle, kokeillen jokaista
141      * huonetta 100 kertaa.
142      */
143     public void sijoitaHuoneet() { // Huoneiden sijoittamisen algoritmi
144         ArrayList<Integer> tempHuoneLista = new ArrayList(huoneLista);
145         this.sijoitetutHuoneet = new ArrayList();
146         Random rng = new Random();
147         while (!tempHuoneLista.isEmpty()) {
148             int sijoitettavaHuone = tempHuoneLista.remove(tempHuoneLista.size() - 1);
149             for (int i = 0; i < 100; i++) { // i kertoo montako kertaa huonetta yritetään sijoittaa tyhjään tilaan.
150                 int x = 1 + rng.nextInt(koko - sijoitettavaHuone - 1); // ei sijoiteta reunoille, jotta käytävä voi alkaa miltä tahansa sivulta
151                 int y = 1 + rng.nextInt(koko - sijoitettavaHuone - 1);
152                 if (kartta[x][y] == 0) {
153                     if (tarkistaOnkoAvoin(x, y, sijoitettavaHuone)) { // tarkistaa onko huoneen paikka pelkästään 0
154                         taytaHuone(x, y, sijoitettavaHuone);
155                         this.huoneidenMaara++;
156                         Huone uusiHuone = new Huone(x, y, sijoitettavaHuone);
157                         this.sijoitetutHuoneet.add(uusiHuone);
158                         break;
159                     }
160                 }
161             }
162         }
163     }

165     /**
166      * Tarkistaa voidaanko tähän tilaan sijoittaa haluttu huone eli mahtuuko se
167      * siihen, ilman että se menee minkään muun huoneen vierelle.
168      *
169      * @param x huoneen vasemman yläkulman x koordinaatti
170      * @param y huoneen vasemman yläkulman y koordinaatti
171      * @param sijoitettavaHuone huoneen sivun leveys.
172      * @return Jos voidaan sijoittaa, palauttaa tosi.
173      */
174     public boolean tarkistaOnkoAvoin(int x, int y, int sijoitettavaHuone) { // Tarkistaa, että huone mahtuu kokonaisuudessaan tilaan
175         int viimeinenX = sijoitettavaHuone + x + 1; // viimeisen tarkistettavan täytyy mennä yli, jotta huoneet eivät ole vierekkäin
176         int aloitusX = x - 1; //viimeisen aloitettavan täytyy alkaa aiemmin samasta syystä
177         if (aloitusX < 0) {
178             aloitusX = 0;
179         }
180         int viimeinenY = sijoitettavaHuone + y + 1;
181         int aloitusY = y - 1;
182         if (aloitusY < 0) {
183             aloitusY = 0;
184         }
185         for (int tarkistettavaY = aloitusY; tarkistettavaY <= viimeinenY && tarkistettavaY < koko; tarkistettavaY++) { // varmista että on taulukon rajoissa
186             for (int tarkistettavaX = aloitusX; tarkistettavaX <= viimeinenX && tarkistettavaX < koko; tarkistettavaX++) {
187                 if (kartta[tarkistettavaY][tarkistettavaX] != 0) {
188                     return false;
189                 }
190             }
191         }
192         return true;
193     }

196      * Täyttää huoneen ykkösillä kartalla.

198      * @param x vasen yläkulma x
199      * @param y vasen yläkulma y
200      * @param sijoitettavaHuone sivun pituus
201      */
202     public void taytaHuone(int x, int y, int sijoitettavaHuone) { //Vaihtaa huoneen paikat kartalla 0:sta 1:ksi
203         for (int muutettavaY = y; muutettavaY < sijoitettavaHuone + y && muutettavaY < koko; muutettavaY++) {
204             for (int muutettavaX = x; muutettavaX < sijoitettavaHuone + x && muutettavaX < koko; muutettavaX++) {
205                 kartta[muutettavaY][muutettavaX] = 1;
206             }
207         }
208     }

211      * Käy läpi sijoitettujen käytävien listaa ja sijoittaa käytävän nykyisen ja

214     public void luoKaytavat() { // Luodaan käytävät huoneiden välillä jotta niissä pystyy liikkumaan
215         sijoitetutHuoneetPoisto = new ArrayList(sijoitetutHuoneet);
216         seuraavaHuone = new Huone(0, 0, 0);
217         if (!sijoitetutHuoneetPoisto.isEmpty()) {
218             nykyinenHuone = sijoitetutHuoneetPoisto.remove(sijoitetutHuoneetPoisto.size() - 1); // Otetaan viimeisin huone
219 //            System.out.println("sijoitetut huoneet poisto size: " + sijoitetutHuoneetPoisto.size());
220             while (!sijoitetutHuoneetPoisto.isEmpty()) { // aloitetaan käymään läpi
221                 if (seuraavaHuone.getHuoneenLeveys() != 0) { // jos ei vielä määritetty seuraavaa huonetta, eli ensimmäinen kerta !!!AIHEUTTI VIAN JOSSA UUDEN KERROKSEN LUONTI EI TOIMI
222                     nykyinenHuone = seuraavaHuone; // muutetaan huone johon luotiin käytävä huoneeksi, josta aloitetaan luomaan käytävä
223                 }
224                 seuraavaHuone = sijoitetutHuoneetPoisto.remove(sijoitetutHuoneetPoisto.size() - 1); // otetaan huone, johon ensimmäisestä luodaan käytävä
225 //                System.out.println("sijoitetut huoneet poisto size: " + sijoitetutHuoneetPoisto.size());
226                 arvoAloitusPaikat();
227             }
228         }

241     /**
242      * Arvotun vaihtoehdon perusteella vaihtaa aloituspaikan koordinaatteja
243      * niin, että ne ovat oikealla sivustalla ja lähtevät liikkumaan oikeaan
244      * suuntaan.
245      *
246      * @param ehto Huoneen sivu, jolta käytävä aloitetaan 0 = huoneen yläosa 1 =
247      * vasen sivu 2 = oikea sivu 3 = alaosa.
248      */
249     public void kayVaihtoehdotLapi(int ehto) {
250         Random rng = new Random();
251         Huone huone = nykyinenHuone;
252         int huoneenLeveys = huone.getHuoneenLeveys();
253         int muutettavaX = huone.getX() - 1;
254         int muutettavaY = huone.getY() - 1;
255         int aloitusSuunta;
256         if (ehto == 1 || ehto == 2) { // Vasen tai oikea sivu
257             aloitusSuunta = 0; // aloitetaan siirtämällä y
258             muutettavaY += 1 + rng.nextInt(huoneenLeveys);
259         } else { // ylös tai alas
260             aloitusSuunta = 1; // aloitetaan siirtämällä x
261             muutettavaX += 1 + rng.nextInt(huoneenLeveys);
262         }
263         if (ehto == 2) {
264             muutettavaX += huoneenLeveys + 1;
265         } else if (ehto == 3) {
266             muutettavaY += huoneenLeveys + 1;
267         }
268         teeKaytava(muutettavaX, muutettavaY, aloitusSuunta);
269     }

271     /**
272      * Luo käytävän kahden pisteen välille, muuttamalla ensin x tai y
273      * koordinaatit oikeaan kohtaan ja sen jälkeen jäljelle jäävän suunnan.
274      *
275      * @param x Käytävän aloituspaikan x-koordinaatti.
276      * @param y Käytävän aloituspaikan y-koordinaatti.
277      * @param aloitusSuunta Suunta, johon käytävää ensin aletaan luomaan, 0 = y
278      * ja 1 = x;
279      */
280     public void teeKaytava(int x, int y, int aloitusSuunta) { // Tehdaan kaytava kahden pisteen valille
281         int muutettavaX = x;
282         int muutettavaY = y;
283         int aloitus = aloitusSuunta;
284         kartta[muutettavaY][muutettavaX] = 1;
285         Random rng = new Random();
286         Huone maaranpaa = this.seuraavaHuone; // otetaan maaranpaaksi seuraava huone
287 //        System.out.println("Aloitus x: " + muutettavaX + " Aloitus y: " + muutettavaY);
288         int paaX = maaranpaa.getX() + rng.nextInt(maaranpaa.getHuoneenLeveys()); // arvotaan jokin x-koordinaatti johon käytävä liitetään
289         int paaY = maaranpaa.getY() + rng.nextInt(maaranpaa.getHuoneenLeveys()); // arvotaan jokin y-koordinaatti johon käytävä liitetään
290 //        System.out.println("Lopetus x: " + paaX + " Lopetus y: " + paaY);
291         while (true) {
292             if (paaX == muutettavaX && paaY == muutettavaY) {
293                 this.kaytavienMaara++;
294                 break;
295             }
296             if (aloitus == 0) {
297                 aloitus = 1;
298                 if (muutettavaY < paaY) { // katsotaan onko lähtö y pienempi vai suurempi, ja liikutaan sen mukaan
299                     while (muutettavaY != paaY) { // jos y pienempi kasvatetaan y kunnes saman arvoinen
300                         muutettavaY++;
301                         kartta[muutettavaY][muutettavaX] = 1;
302                     }
303                 } else {
304                     while (muutettavaY != paaY) { //jos y suurempi vähennetään y kunnes saman arvoinen
305                         muutettavaY--;
306                         kartta[muutettavaY][muutettavaX] = 1;
307                     }
308                 }
309             } else {
310                 aloitus = 0;
311                 if (muutettavaX < paaX) { // sama x:n arvoille y:n jälkeen
312                     while (muutettavaX != paaX) {
313                         muutettavaX++;
314                         kartta[muutettavaY][muutettavaX] = 1;
315                     }
316                 } else {
317                     while (muutettavaX != paaX) {
318                         muutettavaX--;
319                         kartta[muutettavaY][muutettavaX] = 1;
320                     }
321                 }
322             }
323         }
324 //        System.out.println("Paatetty x: " + muutettavaX + " Paatetty y: " + muutettavaY);
325     }

327     /**
328      * Sijoittaa paikat, joista pelaaja aloittaa eli alkuPiste ja portaat alas
329      * eli loppuPiste.
330      */
331     public void sijoitaAlkuJaLoppu() { //sijoittaa portaat, josta pelaaja tuli luolan tähän kerrokseen
332         Random rng = new Random();
333         Huone aloitusHuone = this.sijoitetutHuoneet.get(0);
334         Huone lopetusHuone = this.sijoitetutHuoneet.get(sijoitetutHuoneet.size() - 1);
335         int aloitusX = aloitusHuone.getX() + rng.nextInt(aloitusHuone.getHuoneenLeveys());
336         int aloitusY = aloitusHuone.getY() + rng.nextInt(aloitusHuone.getHuoneenLeveys());
337         this.alkuPiste = new Koordinaatti(aloitusX, aloitusY);
338         int lopetusX = lopetusHuone.getX() + rng.nextInt(lopetusHuone.getHuoneenLeveys());
339         int lopetusY = lopetusHuone.getY() + rng.nextInt(lopetusHuone.getHuoneenLeveys());
340         while (lopetusX == aloitusX && lopetusY == aloitusY) {
341             lopetusX = lopetusHuone.getX() + rng.nextInt(lopetusHuone.getHuoneenLeveys());
342             lopetusY = lopetusHuone.getY() + rng.nextInt(lopetusHuone.getHuoneenLeveys());
343         }
344         this.loppuPiste = new Koordinaatti(lopetusX, lopetusY);
345     }
