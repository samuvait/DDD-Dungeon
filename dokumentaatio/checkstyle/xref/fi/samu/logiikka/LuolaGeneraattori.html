37      public void alustaLuola(int koko, int huoneenLeveysMinimi, int huoneenLeveysMaksimi) {
38          if (koko < 25) {
39              this.koko = 25;  // Kartan minimikoko tällä hetkellä 25
40          } else {
41              this.koko = koko;  // Kartan sivun pituus, vaatii kokeilua oikean löytämiseksi
42          }
43          if (huoneenLeveysMinimi < 3 || huoneenLeveysMinimi > 8) {
44              this.huoneenLeveysMinimi = 3;  // huoneen minimikoko tällä hetkellä 3
45          } else {
46              this.huoneenLeveysMinimi = huoneenLeveysMinimi;
47          }
48          if (huoneenLeveysMaksimi > 8 || huoneenLeveysMaksimi < 3) {
49              this.huoneenLeveysMaksimi = 8;  // huoneen maksimikoko tällä hetkellä 8
50          } else {
51              this.huoneenLeveysMaksimi = huoneenLeveysMaksimi;
52          }
53          this.tayttoAste = (this.koko * this.koko) / 3; //Täyttöasteen määritys, vaikuttaa siihen paljonko kartan pinta-alasta täytetään huoneilla.
54          alustaKartta(this.koko);
55          luoHuoneet();
56          sijoitaHuoneet();
57          luoKaytavat();
58  //        System.out.println("Huoneiden Maara: " + this.huoneidenMaara);
59  //        System.out.println("Kaytavien Maara: " + this.kaytavienMaara);
60      }

62      public void tulostaKartta() {
63          boolean rivinvaihto = false;
64          for (int i = 0; i < koko; i++) {
65              if (rivinvaihto) {
66                  System.out.println("");
67              }
68              for (int j = 0; j < koko; j++) {
69                  System.out.print(kartta[i][j]);
70              }
71              rivinvaihto = true;
72          }
73      }

75      public void alustaKartta(int koko) { // Nollaa kartan kaikki alkiot
76          kartta = new int[koko][koko];
77          for (int i = 0; i < koko; i++) {
78              for (int j = 0; j < koko; j++) {
79                  kartta[i][j] = 0; // Alustetaan kaikki kartan alkiot siten että niihin ei pysty liikkumaan eli 0
80              }
81          }
82      }

84      public void luoHuoneet() { // Luo listan huoneista, jotka sijoitetaan karttaan.
85          Random rng = new Random();
86          huoneLista = new ArrayList();
87          int hlmin = huoneenLeveysMinimi;
88          int hlmax = huoneenLeveysMaksimi;
89          int kaytettyPintaAla = 0;
90          while (kaytettyPintaAla < tayttoAste) {
91              int huoneenLeveys = hlmin + rng.nextInt(hlmax - hlmin); // Huoneiden koon määritys tässä 3-8 leveys ja korkeus.
92              huoneLista.add(huoneenLeveys);
93              kaytettyPintaAla += huoneenLeveys * huoneenLeveys;
94          }
95      }

97      public void sijoitaHuoneet() { // Huoneiden sijoittamisen algoritmi
98          ArrayList<Integer> tempHuoneLista = new ArrayList(huoneLista);
99          this.sijoitetutHuoneet = new ArrayList();
100         Random rng = new Random();
101         while (!tempHuoneLista.isEmpty()) {
102             int sijoitettavaHuone = tempHuoneLista.remove(tempHuoneLista.size() - 1);
103             for (int i = 0; i < 100; i++) { // i kertoo montako kertaa huonetta yritetään sijoittaa tyhjään tilaan.
104                 int x = rng.nextInt(koko - sijoitettavaHuone - 1);
105                 int y = rng.nextInt(koko - sijoitettavaHuone - 1);
106                 if (kartta[x][y] == 0) {
107                     if (tarkistaOnkoAvoin(x, y, sijoitettavaHuone)) { // tarkistaa onko huoneen paikka pelkästään 0
108                         taytaHuone(x, y, sijoitettavaHuone);
109                         this.huoneidenMaara++;
110                         Huone uusiHuone = new Huone(x, y, sijoitettavaHuone);
111                         this.sijoitetutHuoneet.add(uusiHuone);
112                         break;
113                     }
114                 }
115             }
116         }
117     }

119     public boolean tarkistaOnkoAvoin(int x, int y, int sijoitettavaHuone) { // Tarkistaa, että huone mahtuu kokonaisuudessaan tilaan
120         int viimeinenX = sijoitettavaHuone + x + 2; // viimeisen tarkistettavan täytyy mennä yli, jotta huoneet eivät ole vierekkäin
121         int aloitusX = x - 1; //viimeisen aloitettavan täytyy alkaa aiemmin samasta syystä
122         if (aloitusX < 0) {
123             aloitusX = 0;
124             viimeinenX = sijoitettavaHuone + 1; // jos aloitetaan 0
125         }
126         int viimeinenY = sijoitettavaHuone + y + 2;
127         int aloitusY = y - 1;
128         if (aloitusY < 0) {
129             aloitusY = 0;
130             viimeinenY = sijoitettavaHuone + 1;
131         }
132         for (int tarkistettavaX = aloitusX; tarkistettavaX <= viimeinenX && tarkistettavaX < koko; tarkistettavaX++) { // varmista että on taulukon rajoissa
133             for (int tarkistettavaY = aloitusY; tarkistettavaY <= viimeinenY && tarkistettavaY < koko; tarkistettavaY++) {
134                 if (kartta[tarkistettavaX][tarkistettavaY] != 0) {
135                     return false;
136                 }
137             }
138         }
139         return true;
140     }

142     public void taytaHuone(int x, int y, int sijoitettavaHuone) { //Vaihtaa huoneen paikat kartalla 0:sta 1:ksi
143 
144         for (int muutettavaX = x; muutettavaX < sijoitettavaHuone + x && muutettavaX < koko; muutettavaX++) {
145             for (int muutettavaY = y; muutettavaY < sijoitettavaHuone + y && muutettavaY < koko; muutettavaY++) {
146                 kartta[muutettavaX][muutettavaY] = 1;
147             }
148         }
149     }

151     public void luoKaytavat() { // Luodaan käytävät huoneiden välillä jotta niissä pystyy liikkumaan
152         if (sijoitetutHuoneet != null) {
153             nykyinenHuone = sijoitetutHuoneet.remove(sijoitetutHuoneet.size() - 1); // Otetaan viimeisin huone
154             while (!sijoitetutHuoneet.isEmpty()) { // aloitetaan käymään läpi
155                 if (seuraavaHuone != null) { // jos ei vielä määritetty seuraavaa huonetta, eli ensimmäinen kerta
156                     nykyinenHuone = seuraavaHuone; // muutetaan huone johon luotiin käytävä huoneeksi, josta aloitetaan luomaan käytävä
157                 }
158                 seuraavaHuone = sijoitetutHuoneet.remove(sijoitetutHuoneet.size() - 1); // otetaan huone, johon ensimmäisestä luodaan käytävä
159                 arvoAloitusPaikat(nykyinenHuone);
160             }
161         }

164     public void arvoAloitusPaikat(Huone huone) { // Arpoo käytävän aloituspaikat
165         Random rng = new Random();
166         int x = huone.getX();
167         int y = huone.getY();
168         int huoneenLeveys = huone.getHuoneenLeveys();
169         ArrayList<Integer> vaihtoEhdot = new ArrayList();
170         vaihtoEhdot.addAll(Arrays.asList(0, 1, 2, 3)); // luodaan 4 vaihtoehtoa 0 = huoneen yläosa 1 = vasen sivu 2 = oikea sivu 3 = alaosa
171         int ehto = rng.nextInt(4); // Arvotaan neljästä yksi
172         if (y == 0) { // jos y = 0 ei voida sijoittaa yläosaan
173             if (x == 0) { // jos x = 0 ei voida sijoittaa vasemmalle sivulle
174                 ehto = 2 + rng.nextInt(2); // jää 2 ja 3
175             } else {
176                 vaihtoEhdot.remove(0); // poistetaan yläosa
177                 ehto = vaihtoEhdot.get(rng.nextInt(3));
178             }
179         } else if (x == 0) { // Jos vain x = 0
180             vaihtoEhdot.remove(1); // poistetaan vasen sivu valikoimasta
181             ehto = vaihtoEhdot.get(rng.nextInt(3));
182         } else if (y + huoneenLeveys == koko - 1) { // jos huone menee aivan alareunaan
183             if (x + huoneenLeveys == koko - 1) {
184                 ehto = rng.nextInt(2); // ei voida sijoittaa 2 ja 3
185             } else {
186                 vaihtoEhdot.remove(3);
187                 ehto = vaihtoEhdot.get(rng.nextInt(3));
188             }
189         } else if (x + huoneenLeveys == koko - 1) { // jos huone menee oikeaan reunaan
190             vaihtoEhdot.remove(2);
191             ehto = vaihtoEhdot.get(rng.nextInt(3)); // ei voida sijoittaa alaosaan
192         }
193         kayVaihtoehdotLapi(ehto, huone);
194     }

196     public void kayVaihtoehdotLapi(int ehto, Huone huone) {
197         Random rng = new Random();
198         int x = huone.getX();
199         int y = huone.getY();
200         int huoneenLeveys = huone.getHuoneenLeveys();
201         int muutettavaX;
202         int muutettavaY;
203         switch (ehto) { //kaikki tapaukset 0 = huoneen yläosa 1 = vasen sivu 2 = oikea sivu 3 = alaosa
204             case 0:
205                 muutettavaX = x + rng.nextInt(huoneenLeveys - 1);
206                 muutettavaY = y; // System.out.println(x + " " + y + " huone: " + muutettavaX + " " + muutettavaY);
207                 teeKaytavaX(muutettavaX, muutettavaY); // aloitetaan siirtämällä x
208                 break;
209             case 1:
210                 muutettavaX = x;
211                 muutettavaY = y + rng.nextInt(huoneenLeveys - 1); // System.out.println(x + " " + y + " huone: " + muutettavaX + " " + muutettavaY);
212                 teeKaytavaY(muutettavaX, muutettavaY); // aloitetaan siirtämällä y
213                 break;
214             case 2:
215                 muutettavaX = x + huoneenLeveys;
216                 muutettavaY = y + rng.nextInt(huoneenLeveys - 1); // System.out.println(x + " " + y + " huone: " + muutettavaX + " " + muutettavaY);
217                 teeKaytavaY(muutettavaX, muutettavaY);
218                 break;
219             case 3:
220                 muutettavaX = x + rng.nextInt(huoneenLeveys - 1);
221                 muutettavaY = y + huoneenLeveys; // System.out.println(x + " " + y + " huone: " + muutettavaX + " " + muutettavaY);
222                 teeKaytavaX(muutettavaX, muutettavaY);
223                 break;
224         }
225     }

227     public void teeKaytavaY(int x, int y) { // aloitetaan liikuttamalla y
228         this.kaytavienMaara++;
229         int muutettavaX = x;
230         int muutettavaY = y;
231         kartta[muutettavaY][muutettavaX] = 1;
232         Random rng = new Random();
233         Huone maaranpaa = this.seuraavaHuone; // otetaan maaranpaaksi seuraava huone
234         int paaX = maaranpaa.getX() + rng.nextInt(maaranpaa.getHuoneenLeveys() - 1); // arvotaan jokin x-koordinaatti johon käytävä liitetään
235         int paaY = maaranpaa.getY() + rng.nextInt(maaranpaa.getHuoneenLeveys() - 1); // arvotaan jokin y-koordinaatti johon käytävä liitetään
236         while (paaX != muutettavaX && paaY != muutettavaY) { // silloin kun ei olla päätekoordinaateissa
237             if (muutettavaY < paaY) { // katsotaan onko lähtö y pienempi vai suurempi, ja liikutaan sen mukaan
238                 while (muutettavaY != paaY) { // jos y pienempi kasvatetaan y kunnes saman arvoinen
239                     muutettavaY++;
240                     kartta[muutettavaY][muutettavaX] = 1;
241                 }
242             } else {
243                 while (muutettavaY != paaY) { //jos y suurempi vähennetään y kunnes saman arvoinen
244                     muutettavaY--;
245                     kartta[muutettavaY][muutettavaX] = 1;
246                 }
247             }
248             if (muutettavaX < paaX) { // sama x:n arvoille y:n jälkeen
249                 while (muutettavaX != paaX) {
250                     muutettavaX++;
251                     kartta[muutettavaY][muutettavaX] = 1;
252                 }
253             } else {
254                 while (muutettavaX != paaX) {
255                     muutettavaX--;
256                     kartta[muutettavaY][muutettavaX] = 1;
257                 }
258             }
259         }
260     }
