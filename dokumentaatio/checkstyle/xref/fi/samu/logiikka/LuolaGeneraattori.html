43      public LuolaGeneraattori(int koko, int huoneenLeveysMinimi, int huoneenLeveysMaksimi) {
44          if (koko < 10) {
45              this.koko = 10;  // Kartan minimikoko tällä hetkellä 25
46          } else {
47              this.koko = koko;  // Kartan sivun pituus, vaatii kokeilua oikean löytämiseksi
48          }
49          if (huoneenLeveysMinimi < 3 || huoneenLeveysMinimi > 8) {
50              this.huoneenLeveysMinimi = 3;  // huoneen minimikoko tällä hetkellä 3
51          } else {

53          }
54          if (huoneenLeveysMaksimi > 8 || huoneenLeveysMaksimi < 3) {
55              this.huoneenLeveysMaksimi = 8;  // huoneen maksimikoko tällä hetkellä 8
56          } else {
57              this.huoneenLeveysMaksimi = huoneenLeveysMaksimi;
58          }

86      public void alustaKartta(int koko) { // Nollaa kartan kaikki alkiot
87          kartta = new int[koko][koko];
88          for (int i = 0; i < koko; i++) {
89              for (int j = 0; j < koko; j++) {
90                  kartta[i][j] = 0; // Alustetaan kaikki kartan alkiot siten että niihin ei pysty liikkumaan eli 0
91              }
92          }
93      }

95      public void luoHuoneet() { // Luo listan huoneista, jotka sijoitetaan karttaan.
96          Random rng = new Random();
97          huoneLista = new ArrayList();
98          int hlmin = huoneenLeveysMinimi;
99          int hlmax = huoneenLeveysMaksimi;
100         int kaytettyPintaAla = 0;
101         while (kaytettyPintaAla < tayttoAste) {
102             int huoneenLeveys = hlmin + rng.nextInt(1 + hlmax - hlmin); // Huoneiden koon määritys tässä 3-8 leveys ja korkeus.
103             huoneLista.add(huoneenLeveys);
104             kaytettyPintaAla += huoneenLeveys * huoneenLeveys;
105         }
106     }

108     public void sijoitaHuoneet() { // Huoneiden sijoittamisen algoritmi
109         ArrayList<Integer> tempHuoneLista = new ArrayList(huoneLista);
110         this.sijoitetutHuoneet = new ArrayList();
111         Random rng = new Random();
112         while (!tempHuoneLista.isEmpty()) {
113             int sijoitettavaHuone = tempHuoneLista.remove(tempHuoneLista.size() - 1);
114             for (int i = 0; i < 100; i++) { // i kertoo montako kertaa huonetta yritetään sijoittaa tyhjään tilaan.
115                 int x = 1 + rng.nextInt(koko - sijoitettavaHuone - 1); // ei sijoiteta reunoille, jotta käytävä voi alkaa miltä tahansa sivulta
116                 int y = 1 + rng.nextInt(koko - sijoitettavaHuone - 1);
117                 if (kartta[x][y] == 0) {
118                     if (tarkistaOnkoAvoin(x, y, sijoitettavaHuone)) { // tarkistaa onko huoneen paikka pelkästään 0
119                         taytaHuone(x, y, sijoitettavaHuone);
120                         this.huoneidenMaara++;
121                         Huone uusiHuone = new Huone(x, y, sijoitettavaHuone);
122                         this.sijoitetutHuoneet.add(uusiHuone);
123                         break;
124                     }
125                 }
126             }
127         }
128     }

130     public boolean tarkistaOnkoAvoin(int x, int y, int sijoitettavaHuone) { // Tarkistaa, että huone mahtuu kokonaisuudessaan tilaan
131         int viimeinenX = sijoitettavaHuone + x + 1; // viimeisen tarkistettavan täytyy mennä yli, jotta huoneet eivät ole vierekkäin
132         int aloitusX = x - 1; //viimeisen aloitettavan täytyy alkaa aiemmin samasta syystä
133         if (aloitusX < 0) {
134             aloitusX = 0;
135         }
136         int viimeinenY = sijoitettavaHuone + y + 1;
137         int aloitusY = y - 1;
138         if (aloitusY < 0) {
139             aloitusY = 0;
140         }
141         for (int tarkistettavaY = aloitusY; tarkistettavaY <= viimeinenY && tarkistettavaY < koko; tarkistettavaY++) { // varmista että on taulukon rajoissa
142             for (int tarkistettavaX = aloitusX; tarkistettavaX <= viimeinenX && tarkistettavaX < koko; tarkistettavaX++) {
143                 if (kartta[tarkistettavaY][tarkistettavaX] != 0) {
144                     return false;
145                 }
146             }
147         }
148         return true;
149     }

151     public void taytaHuone(int x, int y, int sijoitettavaHuone) { //Vaihtaa huoneen paikat kartalla 0:sta 1:ksi
152         for (int muutettavaY = y; muutettavaY < sijoitettavaHuone + y && muutettavaY < koko; muutettavaY++) {
153             for (int muutettavaX = x; muutettavaX < sijoitettavaHuone + x && muutettavaX < koko; muutettavaX++) {
154                 kartta[muutettavaY][muutettavaX] = 1;
155             }
156         }
157     }

159     public void luoKaytavat() { // Luodaan käytävät huoneiden välillä jotta niissä pystyy liikkumaan
160         sijoitetutHuoneetPoisto = new ArrayList(sijoitetutHuoneet);
161         if (!sijoitetutHuoneetPoisto.isEmpty()) {
162             nykyinenHuone = sijoitetutHuoneetPoisto.remove(sijoitetutHuoneetPoisto.size() - 1); // Otetaan viimeisin huone
163             while (!sijoitetutHuoneetPoisto.isEmpty()) { // aloitetaan käymään läpi
164                 if (seuraavaHuone != null) { // jos ei vielä määritetty seuraavaa huonetta, eli ensimmäinen kerta
165                     nykyinenHuone = seuraavaHuone; // muutetaan huone johon luotiin käytävä huoneeksi, josta aloitetaan luomaan käytävä
166                 }
167                 seuraavaHuone = sijoitetutHuoneetPoisto.remove(sijoitetutHuoneetPoisto.size() - 1); // otetaan huone, johon ensimmäisestä luodaan käytävä
168                 arvoAloitusPaikat(nykyinenHuone);
169             }
170         }
171     }

174         Random rng = new Random();
175         int ehto = rng.nextInt(4); // luodaan 4 vaihtoehtoa 0 = huoneen yläosa 1 = vasen sivu 2 = oikea sivu 3 = alaosa
176         kayVaihtoehdotLapi(ehto, huone);

179     public void kayVaihtoehdotLapi(int ehto, Huone huone) {
180         Random rng = new Random();
181         int huoneenLeveys = huone.getHuoneenLeveys();
182         int muutettavaX = huone.getX();
183         int muutettavaY = huone.getY();
184         int aloitusSuunta;
185         if (ehto == 1 || ehto == 2) { // Vasen tai oikea sivu
186             aloitusSuunta = 0; // aloitetaan siirtämällä y
187             muutettavaY += rng.nextInt(huoneenLeveys - 1);
188         } else { // ylös tai alas
189             aloitusSuunta = 1; // aloitetaan siirtämällä x
190             muutettavaX += rng.nextInt(huoneenLeveys - 1);
191         }
192         if (ehto == 2) {
193             muutettavaX += huoneenLeveys;
194         } else if (ehto == 3) {
195             muutettavaY += huoneenLeveys;
196         }
197         teeKaytava(muutettavaX, muutettavaY, aloitusSuunta);
198     }

200     public void teeKaytava(int x, int y, int aloitusSuunta) { // Tehdaan kaytava kahden pisteen valille
201         int muutettavaX = x;
202         int muutettavaY = y;
203         int aloitus = aloitusSuunta;
204         kartta[muutettavaY][muutettavaX] = 1;
205         Random rng = new Random();
206         Huone maaranpaa = this.seuraavaHuone; // otetaan maaranpaaksi seuraava huone
207         int paaX = maaranpaa.getX() + rng.nextInt(maaranpaa.getHuoneenLeveys()); // arvotaan jokin x-koordinaatti johon käytävä liitetään
208         int paaY = maaranpaa.getY() + rng.nextInt(maaranpaa.getHuoneenLeveys()); // arvotaan jokin y-koordinaatti johon käytävä liitetään
209         while (true) {
210             if (paaX == muutettavaX && paaY == muutettavaY) {
211                 this.kaytavienMaara++;
212                 break;
213             }
214             if (aloitus == 0) {
215                 aloitus = 1;
216                 if (muutettavaY < paaY) { // katsotaan onko lähtö y pienempi vai suurempi, ja liikutaan sen mukaan
217                     while (muutettavaY != paaY) { // jos y pienempi kasvatetaan y kunnes saman arvoinen
218                         muutettavaY++;
219                         kartta[muutettavaY][muutettavaX] = 1;
220                     }
221                 } else {
222                     while (muutettavaY != paaY) { //jos y suurempi vähennetään y kunnes saman arvoinen
223                         muutettavaY--;
224                         kartta[muutettavaY][muutettavaX] = 1;
225                     }
226                 }
227             } else {
228                 aloitus = 0;
229                 if (muutettavaX < paaX) { // sama x:n arvoille y:n jälkeen
230                     while (muutettavaX != paaX) {
231                         muutettavaX++;
232                         kartta[muutettavaY][muutettavaX] = 1;
233                     }
234                 } else {
235                     while (muutettavaX != paaX) {
236                         muutettavaX--;
237                         kartta[muutettavaY][muutettavaX] = 1;
238                     }
239                 }
240             }
241         }
242     }

244     public void sijoitaAlkuJaLoppu() { //sijoittaa portaat, josta pelaaja tuli luolan tähän kerrokseen
245         Random rng = new Random();
246         Huone aloitusHuone = this.sijoitetutHuoneet.get(0);
247         Huone lopetusHuone = this.sijoitetutHuoneet.get(sijoitetutHuoneet.size() - 1);
248         int aloitusX = aloitusHuone.getX() + rng.nextInt(aloitusHuone.getHuoneenLeveys());
249         int aloitusY = aloitusHuone.getY() + rng.nextInt(aloitusHuone.getHuoneenLeveys());
250         this.alkuPiste = new Koordinaatti(aloitusX, aloitusY);
251         int lopetusX = lopetusHuone.getX() + rng.nextInt(lopetusHuone.getHuoneenLeveys());
252         int lopetusY = lopetusHuone.getY() + rng.nextInt(lopetusHuone.getHuoneenLeveys());
253         this.loppuPiste = new Koordinaatti(lopetusX, lopetusY);
254     }
