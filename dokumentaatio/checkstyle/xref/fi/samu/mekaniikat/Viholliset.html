11  /**
12   * Luokka luo ja sijoittaa hirviöt joita vastaan pelaaja taistelee, sekä

25      /**
26       * Luo vihollisten esiintymät kerrokseen.
27       *
28       * @param koko Luolan koko.
29       * @param kerros Luolan kerros, mikä määrittää vihollisten vaikeuden.
30       * @param sijoitetutHuoneet Huoneet jotka on sijoitettu, ja joihin
31       * sijoitetaan otuksia.
32       * @param kartta Kartta, johon sijoitetaan otuksia.
33       */
34      public Viholliset(int koko, int kerros, ArrayList<Huone> sijoitetutHuoneet, int[][] kartta) {
35          this.kartta = kartta;
36          this.kerros = kerros;
37          this.sijoitetutHuoneet = new ArrayList(sijoitetutHuoneet);
38          this.koko = koko;
39      }

41      /**
42       * Luo tietyn kerroksen viholliset. Mitä suurempi kerros, sitä voimakkaampia
43       * vihollisia voi syntyä.
44       */
45      public void luoViholliset() {
46          viholliset = new ArrayList();
47          Random rng = new Random();
48          this.sijoitetutHuoneet.remove(0);
49          varmistus = sijoitetutHuoneet.size();
50          lkm = 0;
51          int tyyppiMax = kerros;
52          if (tyyppiMax > 1) {
53              tyyppiMax = 1;
54          }
55          for (int i = 0; i < sijoitetutHuoneet.size(); i++) {
56              Huone otuksenHuone = sijoitetutHuoneet.get(i);
57              int tyyppi = rng.nextInt(tyyppiMax);
58              int otusX = otuksenHuone.getX() + rng.nextInt(otuksenHuone.getHuoneenLeveys());
59              int otusY = otuksenHuone.getY() + rng.nextInt(otuksenHuone.getHuoneenLeveys());
60              if (kartta[otusY][otusX] == 1) {
61                  Otus lisattava = new Otus(otusX, otusY, tyyppi);
62                  viholliset.add(lisattava);
63                  lkm++;
64              }
65          }
66          //System.out.println(varmistus + " =? " + lkm);
67      }

69      /**
70       * Sijoittaa viholliset kartalle tekstikäyttöliittymää varten.
71       */
72      public void sijoitaViholliset() {
73          if (viholliset != null) {
74              for (Otus otus : viholliset) {
75                  int x = otus.getKoordinaatit().getX();
76                  int y = otus.getKoordinaatit().getY();
77                  kartta[y][x] = 6;
78              }
79          }
80      }

82      /**
83       * Tarkistaa, onko paikassa johon pelaaja haluaa liikkua hirviö ja jos on,
84       * taistelee sitä vastaan.
85       *
86       * @param x Liikkeen x-koordinaatti.
87       * @param y Liikkeen y-koordinaatti.
88       * @return Palauttaa tosi, jos pelaaja on taistellut jotain hirviötä
89       * vastaan.
90       */
91      public boolean tarkista(int x, int y) {
92          boolean ret = false;
93          if (viholliset != null) {
94              for (Otus otus : viholliset) {
95                  if (otus.getKoordinaatit().getX() == x && otus.getKoordinaatit().getY() == y) {
96                      otus.setTaisteleeko(1);
97                      ret = true;
98                      break;
99                  }
100             }
101         }
102         return ret;
103     }

105     /**
106      * Tarkistaa onko paikassa, johon Otus haluaa liikkua jo toinen otus.
107      *
108      * @param x Liikuttava paikan x-koordinaatti.
109      * @param y Liikuttavan paikan y-koordinaatti.
110      * @return Palauttaa true, jos toinen otus on ja ei voida liikkua.
111      */
112     public boolean tarkistaOtukset(int x, int y) {
113         boolean ret = false;
114         if (viholliset != null) {
115             for (Otus otus : viholliset) {
116                 if (otus.getKoordinaatit().getX() == x && otus.getKoordinaatit().getY() == y) {
117                     ret = true;
118                     break;
119                 }
120             }
121         }
122         return ret;
123     }

125     /**
126      * Otukset jotka on määritetty taistelemaan pelaajaa vastaan, taistelevat ja
127      * satuttavat pelaajaa, sekä ottavat vahinkoa.
128      *
129      * @param pelaaja Pelaaja, jota vastaan taistellaan.
130      */
131     public void taistele(Pelaaja pelaaja) {
132         if (viholliset != null) {
133             for (Otus otus : viholliset) {
134                 if (otus.getTaisteleeko() == 1) {
135                     int hp = pelaaja.getHitPoints();
136                     int ap = pelaaja.getAttackPower();
137                     int otusAP = otus.getAttackPower();
138                     int otusHP = otus.getHitPoints();
139                     hp -= otusAP;
140                     otusHP -= ap;
141                     otus.setHitPoints(otusHP);
142                     pelaaja.setHitPoints(hp);
143                     pelaaja.growExperience(20);
144                     otus.setTaisteleeko(0);
145                 }
146             }
147         }
148     }

150     /**
151      * Tarkistaa kaikki otukset joiden hp on vähemmän kuin 1 ja poistaa ne
152      * pelistä, koska ne ovat kuolleet.
153      *
154      * @return Varmistaa onko yhtäkään otusta poistettu.
155      */
156     public boolean tarkistaKuoliko() {
157         boolean ret = false;
158         for (Iterator<Otus> iterator = this.viholliset.iterator(); iterator.hasNext();) {
159             Otus otus = iterator.next();
160             if (otus.getHitPoints() < 1) {
161                 iterator.remove();
162                 kartta[otus.getKoordinaatit().getY()][otus.getKoordinaatit().getX()] = 1;
163                 System.out.println("You have defeated a " + otus.getKuvaus() + "!");
164                 ret = true;
165             }
166         }
167         return ret;
168     }

170     /**
171      * Liikuttaa kaikkia otuksia, joiden vuoro on liikkua.
172      *
173      * @param pelaaja Pelaaja, jotta otukset tietävät voivatko ne liikkua
174      * siihen.
175      */
176     public void liikuta(Pelaaja pelaaja) {
177         ArrayList<Otus> liikutettavat = new ArrayList();
178         for (Otus otus : viholliset) {
179             if (otus.getLiikkuuko() == 0) {
180                 otus.setLiikkuuko(1);
181             } else {
182                 liikutettavat.add(otus);
183             }
184         }
185         for (Iterator<Otus> iterator = liikutettavat.iterator(); iterator.hasNext();) {
186             Otus otus = iterator.next();
187 //            System.out.println(otus.getKuvaus() + " moved");
188             liiku(otus, pelaaja);
189         }
190     }

192     /**
193      * Liikuttaa otusta pelaajan liikkumisen jälkeen.
194      *
195      * @param otus Otus jota liikutetaan.
196      * @param pelaaja Pelaaja, jotta otus tietää osuuko se pelaajaan.
197      */
198     public void liiku(Otus otus, Pelaaja pelaaja) { // 0 = ylos 1 = vasen 2 = oikea 3 = alas
199         Koordinaatti otusKoor = otus.getKoordinaatit();
200         int x = otusKoor.getX();
201         int y = otusKoor.getY();
202         int pelX = pelaaja.getKoordinaatit().getX();
203         int pelY = pelaaja.getKoordinaatit().getY();
204         Random rng = new Random();
205         ArrayList<Integer> vaihtoEhdot = new ArrayList(asList(0, 1, 2, 3));
206         kartta[y][x] = 1; // tekstikäyttöliittymän line
207         if (abs(pelX - x) < 2 && abs(pelY - y) < 2) {
208             if (otus.voiTaistella(pelX, pelY)) {
209                 otus.setTaisteleeko(1);
210                 System.out.println("The " + otus.getKuvaus() + " attacks!");
211                 this.taistele(pelaaja);
212             }
213         } else {
214             while (true) {
215                 int suunta = vaihtoEhdot.get(rng.nextInt(vaihtoEhdot.size()));
216                 if (suunta == 0 && y > 0 && kartta[y - 1][x] == 1) {
217                     y--;
218                     if (tarkistaOtukset(x, y)) {
219                         y++;
220                     } else {
221                         break;
222                     }
223                 } else if (suunta == 1 && x > 0 && kartta[y][x - 1] == 1) {
224                     x--;
225                     if (tarkistaOtukset(x, y)) {
226                         x++;
227                     } else {
228                         break;
229                     }
230                 } else if (suunta == 2 && x < koko - 1 && kartta[y][x + 1] == 1) {
231                     x++;
232                     if (tarkistaOtukset(x, y)) {
233                         x--;
234                     } else {
235                         break;
236                     }
237                 } else if (suunta == 3 && y < koko - 1 && kartta[y + 1][x] == 1) {
238                     y++;
239                     if (tarkistaOtukset(x, y)) {
240                         y--;
241                     } else {
242                         break;
243                     }
244                 }
245                 poistaSuunta(vaihtoEhdot, suunta);
246                 if (vaihtoEhdot.isEmpty()) {
247                     break;
248                 }
249             }
250         }
251         if (otus.getHitPoints() < 1) {
252             tarkistaKuoliko();
253         } else {
254             kartta[y][x] = 6; // tekstikäyttöliittymän line
255             otus.setLiikkuuko(0);
256             otus.setKoordinaatit(new Koordinaatti(x, y));
257         }
258     }

263      *
264      * @param vaihtoEhdot Vaihtoehdot, joihin otus voi liikkua.

266      */
267     public void poistaSuunta(ArrayList<Integer> vaihtoEhdot, int suunta) {
268         for (Iterator<Integer> iterator = vaihtoEhdot.iterator(); iterator.hasNext();) {
269             int i = iterator.next();
270             if (i == suunta) {
271                 iterator.remove();
272             }
273         }
274     }
