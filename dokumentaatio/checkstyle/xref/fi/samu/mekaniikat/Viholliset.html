4   import fi.samu.logiikka.Huone;

10  /**
11   * Luokka luo ja sijoittaa hirviöt joita vastaan pelaaja taistelee, sekä
12   * tarkistaa osuuko pelaaja niihin ja hoitaa niiden liikkeen.
13   */
14  public class Viholliset {
15  
16      private int kerros;
17      private int koko;
18      private ArrayList<Otus> viholliset;
19      private ArrayList<Huone> sijoitetutHuoneet;
20      private int[][] kartta;
21      public int varmistus;
22      public int lkm;
23      private Taisteleminen taisto;
24      private final int otustenMaara = 1;
25  
26      /**
27       * Luo vihollisten esiintymät kerrokseen.
28       *
29       * @param koko Luolan koko.
30       * @param kerros Luolan kerros, mikä määrittää vihollisten vaikeuden.
31       * @param sijoitetutHuoneet Huoneet jotka on sijoitettu, ja joihin
32       * sijoitetaan otuksia.
33       * @param kartta Kartta, johon sijoitetaan otuksia.
34       */
35      public Viholliset(int koko, int kerros, ArrayList<Huone> sijoitetutHuoneet, int[][] kartta) {
36          this.kartta = kartta;
37          this.kerros = kerros;
38          this.sijoitetutHuoneet = new ArrayList(sijoitetutHuoneet);
39          this.koko = koko;
40      }
41  
42      /**
43       * Luo tietyn kerroksen viholliset. Mitä suurempi kerros, sitä voimakkaampia
44       * vihollisia voi syntyä.
45       */
46      public void luoViholliset() {
47          viholliset = new ArrayList();
48          Random rng = new Random();
49          this.sijoitetutHuoneet.remove(0);
50          varmistus = sijoitetutHuoneet.size();
51          lkm = 0;
52          int tyyppiMax = kerros;
53          if (tyyppiMax > otustenMaara) {
54              tyyppiMax = 1;
55          }
56          for (int i = 0; i < sijoitetutHuoneet.size(); i++) {
57              Huone otuksenHuone = sijoitetutHuoneet.get(i);
58              int tyyppi = rng.nextInt(tyyppiMax);
59              int otusX = otuksenHuone.getX() + rng.nextInt(otuksenHuone.getHuoneenLeveys());
60              int otusY = otuksenHuone.getY() + rng.nextInt(otuksenHuone.getHuoneenLeveys());
61              if (kartta[otusY][otusX] == 1) {
62                  Otus lisattava = new Otus(otusX, otusY, tyyppi);
63                  viholliset.add(lisattava);
64                  kartta[otusY][otusX] = 6;
65                  lkm++;
66              }
67          }
68      }
69  
70      /**
71       * Palauttaa otuksen, joka on kohdassa x ja y.
72       *
73       * @param x Otuksen x-koordinaatti.
74       * @param y Otuksen y-koordinaatti.
75       * @return Palautettava otus.
76       */
77      public Otus palautaPiirrettava(int x, int y) {
78          Otus uusi = new Otus(0, 0, 1);
79          for (Otus otus : viholliset) {
80              if (otus.getKoordinaatit().getX() == x && otus.getKoordinaatit().getY() == y) {
81                  uusi = otus;
82              }
83          }
84          return uusi;
85      }
86  
87      /**
88       * Tarkistaa onko paikassa, johon Otus haluaa liikkua jo toinen otus.
89       *
90       * @param x Liikuttava paikan x-koordinaatti.
91       * @param y Liikuttavan paikan y-koordinaatti.
92       * @return Palauttaa true, jos toinen otus on ja ei voida liikkua.
93       */
94      public boolean tarkistaOtukset(int x, int y) {
95          boolean ret = false;
96          if (viholliset != null) {
97              for (Otus otus : viholliset) {
98                  if (otus.getKoordinaatit().getX() == x && otus.getKoordinaatit().getY() == y) {
99                      ret = true;
100                     break;
101                 }
102             }
103         }
104         return ret;
105     }
106 
107     /**
108      * Liikuttaa kaikkia otuksia, joiden vuoro on liikkua.
109      *
110      * @param pelaaja Pelaaja, jotta otukset tietävät voivatko ne liikkua
111      * siihen.
112      */
113     public void liikuta(Pelaaja pelaaja) {
114         ArrayList<Otus> liikutettavat = new ArrayList();
115         for (Otus otus : viholliset) {
116             if (otus.getLiikkuuko() == 0) {
117                 otus.setLiikkuuko(1);
118             } else {
119                 liikutettavat.add(otus);
120             }
121         }
122         for (Iterator<Otus> iterator = liikutettavat.iterator(); iterator.hasNext();) {
123             Otus otus = iterator.next();
124             this.liiku(otus, pelaaja);
125         }
126     }
127 
128     /**
129      * Liikuttaa otusta pelaajan liikkumisen jälkeen.
130      *
131      * @param otus Otus jota liikutetaan.
132      * @param pelaaja Pelaaja, jotta otus tietää osuuko se pelaajaan.
133      */
134     public void liiku(Otus otus, Pelaaja pelaaja) { // 0 = ylos 1 = vasen 2 = oikea 3 = alas
135         int x = otus.getKoordinaatit().getX();
136         int y = otus.getKoordinaatit().getY();
137         Random rng = new Random();
138         ArrayList<Integer> vaihtoEhdot = new ArrayList(asList(0, 1, 2, 3));
139         kartta[y][x] = 1; // tekstikäyttöliittymän line
140         boolean taisteli = taisto.voikoTaistella(otus, pelaaja);
141         if (!taisteli) {
142             while (true) {
143                 int suunta = vaihtoEhdot.get(rng.nextInt(vaihtoEhdot.size()));
144                 if (suunta == 0 && y > 0 && kartta[y - 1][x] == 1 && !tarkistaOtukset(x, y - 1)) {
145                     y--;
146                     break;
147                 } else if (suunta == 1 && x > 0 && kartta[y][x - 1] == 1 && !tarkistaOtukset(x - 1, y)) {
148                     x--;
149                     break;
150                 } else if (suunta == 2 && x < koko - 1 && kartta[y][x + 1] == 1 && !tarkistaOtukset(x + 1, y)) {
151                     x++;
152                     break;
153                 } else if (suunta == 3 && y < koko - 1 && kartta[y + 1][x] == 1 && !tarkistaOtukset(x, y + 1)) {
154                     y++;
155                     break;
156                 }
157                 poistaSuunta(vaihtoEhdot, suunta);
158                 if (vaihtoEhdot.isEmpty()) {
159                     break;
160                 }
161             }
162         }
163         if (otus.getHitPoints() < 1) {
164             taisto.tarkistaKuoliko();
165         } else {
166             kartta[y][x] = 6; // tekstikäyttöliittymän line
167             otus.setLiikkuuko(0);
168             otus.setKoordinaatit(new Koordinaatti(x, y));
169         }
170     }

172     /**
173      * Poistaa annetun suunnan vaihtoehdoista, jotta voidaan valita oikea
174      * suunta.
175      *
176      * @param vaihtoEhdot Vaihtoehdot, joihin otus voi liikkua.
177      * @param suunta Poistettava vaihtoehto.
178      */
179     public void poistaSuunta(ArrayList<Integer> vaihtoEhdot, int suunta) {
180         for (Iterator<Integer> iterator = vaihtoEhdot.iterator(); iterator.hasNext();) {
181             int i = iterator.next();
182             if (i == suunta) {
183                 iterator.remove();
184             }
185         }
186     }
187 
188     public ArrayList<Otus> getVihollisLista() {
189         return this.viholliset;
190     }
191 
192     public void setTaistelu(Taisteleminen taistelu) {
193         this.taisto = taistelu;
194     }
