6   /**
7    * Luokka kuvaa hirviöitä, joita vastaan pelaaja taistelee.
8    */
9   public class Otus {
10  
11      private Koordinaatti koordinaatit;
12      private int hitPoints;
13      private int attackPower;
14      private int tyyppi;
15      private String kuvaus;
16      private int liikkuuko;
17      private int taisteleeko;
18  
19      /**
20       * Antaa luotaessa otuksen koordinaatit, sekä sen tyypin.
21       *
22       * @param x Otuksen x-koordinaatti.
23       * @param y Otuksen y-koordinaatti.
24       * @param tyyppi Otuksen tyyppi eli sen vahvuus eli ap ja hp.
25       */
26      public Otus(int x, int y, int tyyppi) {
27          int uusiX = x;
28          if (x < 0) {
29              uusiX = 0;
30          }
31          int uusiY = y;
32          if (y < 0) {
33              uusiY = 0;
34          }
35          koordinaatit = new Koordinaatti(uusiX, uusiY);
36          liikkuuko = 0;
37          taisteleeko = 0;
38          if (tyyppi == 0) {
39              this.hitPoints = 5;
40              this.attackPower = 1;
41              this.kuvaus = "rat";
42          } else {
43              this.hitPoints = 5;
44              this.attackPower = 1;
45              this.kuvaus = "rat";
46          }
47      }
48  
49      /**
50       * Tarkistaa, voiko otus taistella eli onko se koordinaattien vieressä,
51       * jotka on annettu.
52       *
53       * @param x x-koordinaatti, joka tarkastetaan.
54       * @param y y-koordinaatti, joka tarkastetaan.
55       * @return Jos voi taistella, eli on vieressä, palauttaa tosi.
56       */
57      public boolean voiTaistella(int x, int y) {
58          boolean ret = false;
59          int otusX = this.getKoordinaatit().getX();
60          int otusY = this.getKoordinaatit().getY();
61          if (abs(otusX - x) == 1 && otusY - y == 0) {
62              ret = true;
63          } else if (abs(otusY - y) == 1 && otusX - x == 0) {
64              ret = true;
65          }
66          return ret;
67      }
68

73       */
74      public void setKoordinaatit(Koordinaatti krdn) {
75          koordinaatit.setX(krdn.getX());
76          koordinaatit.setY(krdn.getY());
77      }
