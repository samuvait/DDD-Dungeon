19      public Otus(int x, int y, int tyyppi) {
20          int uusiX = x;
21          if (x < 0) {
22              uusiX = 0;
23          }
24          int uusiY = y;
25          if (y < 0) {
26              uusiY = 0;
27          }
28          koordinaatit = new Koordinaatti(uusiX, uusiY);
29          liikkuuko = 0;
30          taisteleeko = 0;
31          if (tyyppi == 0) {
32              this.hitPoints = 5;
33              this.attackPower = 1;
34              this.kuvaus = "rotta";
35          } else {
36              this.hitPoints = 5;
37              this.attackPower = 1;
38              this.kuvaus = "rotta";
39          }
40      }
