43      private JPanel luoRuudukkoPaneeli() {
44          this.list = new ArrayList();
45          JPanel p = new JPanel(new GridLayout(sivunPituus, sivunPituus));
46          for (int i = 0; i < sivunPituus * sivunPituus; i++) {
47              JPanel jp = new JPanel();
48              jp.setPreferredSize(new Dimension(dmn, dmn));
49              Border border = BorderFactory.createLineBorder(Color.GRAY);
50              jp.setBorder(border);
51              list.add(jp);
52              p.add(jp);
53          }
54          paivita(new ArrayList<String>());
55          return p;
56      }

58      private JPanel luoRuudukkoBagPaneeli() {
59          this.list = new ArrayList();
60          JPanel p = new JPanel(new GridBagLayout());
61          for (int i = 0; i < sivunPituus; i++) {
62              for (int j = 0; j < sivunPituus; j++) {
63                  JPanel jp = new JPanel();
64                  jp.setPreferredSize(new Dimension(dmn, dmn));
65                  Border border = BorderFactory.createLineBorder(Color.GRAY);
66                  jp.setBorder(border);
67                  GridBagConstraints c = new GridBagConstraints();
68                  c.gridx = j;
69                  c.gridy = i;
70                  list.add(jp);
71                  p.add(jp, c);
72              }
73          }
74          JPanel jp = new JPanel();
75          jp.setLayout(new BoxLayout(jp, BoxLayout.Y_AXIS));
76          jp.setPreferredSize(new Dimension(200, dmn * sivunPituus));
77          GridBagConstraints c = new GridBagConstraints();
78  //        c.fill = GridBagConstraints.BOTH;
79  //        c.weightx = 5;
80          c.gridx = 25;
81          c.gridy = 0;
82          c.weightx = 1;
83          c.gridheight = sivunPituus;
84          p.add(jp, c);
85          combatLog = jp;
86          lisaaLabelit();
87          paivita(new ArrayList<String>());
88          return p;
89      }

103     public void MuutaLabelit(ArrayList<String> tekstit) {
104         ArrayList<String> muutettavat = new ArrayList<String>();
105         for (int i = 46; i >= 0; i--) {
106             JLabel lbl = lblLista.get(i);
107             String muut = lbl.getText();
108             muutettavat.add(muut);
109         }
110         int i = tekstit.size() - 1;
111         int h = 46;
112         while (i >= 0 && h >= 0) {
113             String s = tekstit.get(i);
114             JLabel lbl = lblLista.get(h);
115             lbl.setText(s);
116             h--;
117             i--;
118         }
119         int j = 0;
120         while (h >= 0 && j < muutettavat.size()) {
121             String s = muutettavat.get(j);
122             JLabel lbl = lblLista.get(h);
123             lbl.setText(s);
124             h--;
125             j++;
126         }

129     public void paivita(ArrayList<String> tekstit) {
130         kartta = lg.getKartta();
131         viholliset = liikkuminen.getViholliset();
132         String kerros = "" + liikkuminen.getKerros();
133         f.invalidate();
134         for (int i = 0; i < sivunPituus; i++) {
135             for (int j = 0; j < sivunPituus; j++) {
136                 int index = (j * sivunPituus) + i;
137                 JPanel pan = list.get(index);
138                 pan.removeAll();
139                 if (i == 0 && j == 0) {
140                     pan.setBackground(Color.gray);
141                     Font font = new Font("Courier", Font.PLAIN, 8);
142                     JLabel lbl = new JLabel("Floor");
143                     lbl.setFont(font);
144                     lbl.setBorder(BorderFactory.createEmptyBorder(-3 /*top*/, 0, 0, 0));
145                     JLabel krs = new JLabel(kerros);
146                     krs.setVerticalTextPosition(JLabel.BOTTOM);
147                     Font f = new Font("Courier", Font.PLAIN, 10);
148                     krs.setFont(f);
149                     pan.add(lbl);
150                     pan.add(krs);
151                 } else if (kartta[j][i] == 0) {
152                     pan.setBackground(Color.black);
153                 } else {
154                     pan.setBackground(Color.white);
155                     Font font = new Font("Courier", Font.BOLD, 8);
156                     Font hpFont = new Font("Courier", Font.PLAIN, 8);
157                     Pelaaja p = this.liikkuminen.getPelaaja();
158                     if (i == p.getKoordinaatit().getX() && j == p.getKoordinaatit().getY()) {
159                         JLabel lbl = new JLabel("@");
160                         lbl.setFont(font);
161                         lbl.setBorder(BorderFactory.createEmptyBorder(-3 /*top*/, 0, 0, 0));
162                         JLabel hp = new JLabel(p.getHitPoints() + "/" + p.getHitPointsMax());
163                         hp.setFont(hpFont);
164                         hp.setVerticalTextPosition(JLabel.BOTTOM);
165                         pan.add(lbl);
166                         pan.add(hp);
167                     } else if (this.viholliset.tarkistaOtukset(i, j)) {
168                         Otus o = this.viholliset.palautaPiirrettava(i, j);
169                         JLabel lbl = new JLabel(o.getTunnus());
170                         lbl.setFont(font);
171                         lbl.setBorder(BorderFactory.createEmptyBorder(-3 /*top*/, 0, 0, 0));
172                         JLabel hp = new JLabel(o.getHitPoints() + "/" + o.getHitPointsMax());
173                         hp.setFont(hpFont);
174                         hp.setVerticalAlignment(JLabel.BOTTOM);
175                         pan.add(lbl);
176                         pan.add(hp);
177                     } else {
178                         if (kartta[j][i] == 5) {
179                             JLabel lbl = new JLabel("%");
180                             pan.add(lbl);
181                         }
182                     }
183                 }
184             }
185         }
186         if (tekstit.size() > 0) {
187             MuutaLabelit(tekstit);
188         }
189         f.revalidate();
190         f.repaint();
191         this.kuoliko();
192         this.lvlUp();
193         
194     }

212     public void kuoliko() {
213         Object[] vaihtoEhdot = {
214             "Restart",
215             "Give up"
216         };
217         Pelaaja p = this.liikkuminen.getPelaaja();
218         if (p.getHitPoints() < 1) {
219             int reply = JOptionPane.showOptionDialog(f, "You have died! Do you want to restart?", "Death", JOptionPane.YES_NO_OPTION, JOptionPane.PLAIN_MESSAGE, null, vaihtoEhdot, vaihtoEhdot[0]);
220             if (reply == 0) {
221                 lg = new LuolaGeneraattori(sivunPituus);
222                 liikkuminen = new Liikkuminen(lg);
223                 viholliset = liikkuminen.getViholliset();
224                 kartta = lg.getKartta();
225                 k.setLiikkuminen(liikkuminen);
226                 this.tyhjennaLogi();
227                 this.lisaaLabelit();
228                 this.paivita(new ArrayList<String>());
229             } else {
230                f.dispatchEvent(new WindowEvent(f, WindowEvent.WINDOW_CLOSING));
231             }
232        }
233     }

241     public void lvlUp() {
242         Pelaaja p = this.liikkuminen.getPelaaja();
243         if (p.getExperience() > 99) {
244             p.growExperience(-100);
245             Object[] vaihtoEhdot = {
246                 "Health + 5",
247                 "Health Refill",
248                 "Attack Power + 2"
249             };
250             int reply = JOptionPane.showOptionDialog(f, "Level up! You can choose from 3 options:", "lvlup", JOptionPane.YES_NO_CANCEL_OPTION, JOptionPane.PLAIN_MESSAGE, null, vaihtoEhdot, vaihtoEhdot[0]);
251             int hp = p.getHitPoints();
252             int hpMax = p.getHitPointsMax();
253             if (reply == 0) {
254                 hpMax += 5;
255                 hp += 5;
256                 p.setHitPoints(hp);
257                 p.setHitPointsMax(hpMax);
258             } else if (reply == 1) {
259                 hp = hpMax;
260                 p.setHitPoints(hp);
261             } else {
262                 int ap = p.getAttackPower();
263                 p.setAttackPower(ap + 2);
264             }
265             this.paivita(new ArrayList<String>());
266         }
