47      private JPanel luoRuudukkoBagPaneeli() {
48          this.list = new ArrayList();
49          JPanel p = new JPanel(new GridBagLayout());
50          for (int i = 0; i < sivunPituus; i++) {
51              for (int j = 0; j < sivunPituus; j++) {
52                  JPanel jp = new JPanel();
53                  jp.setPreferredSize(new Dimension(dmn, dmn));
54                  Border border = BorderFactory.createLineBorder(Color.GRAY);
55                  jp.setBorder(border);
56                  GridBagConstraints c = new GridBagConstraints();
57                  c.gridx = j;
58                  c.gridy = i;
59                  list.add(jp);
60                  p.add(jp, c);
61              }
62          }
63          JPanel jp = new JPanel();
64          jp.setLayout(new BoxLayout(jp, BoxLayout.Y_AXIS));
65          jp.setPreferredSize(new Dimension(300, dmn * sivunPituus));
66          GridBagConstraints c = new GridBagConstraints();
67  //        c.fill = GridBagConstraints.BOTH;
68  //        c.weightx = 5;
69          c.gridx = 25;
70          c.gridy = 0;
71          c.weightx = 1;
72          c.gridheight = sivunPituus;
73          p.add(jp, c);
74          combatLog = jp;
75          lisaaLabelit();
76          paivita(new ArrayList<String>());
77          return p;
78      }

99      public void MuutaLabelit(ArrayList<String> tekstit) {
100         ArrayList<String> muutettavat = new ArrayList<String>();
101         for (int i = rivienMaara - 3; i >= 0; i--) {
102             JLabel lbl = lblLista.get(i);
103             String muut = lbl.getText();
104             muutettavat.add(muut);
105         }
106         int i = tekstit.size() - 1;
107         int h = rivienMaara - 3;
108         while (i >= 0 && h >= 0) {
109             String s = tekstit.get(i);
110             JLabel lbl = lblLista.get(h);
111             lbl.setText(s);
112             h--;
113             i--;
114         }
115         int j = 0;
116         while (h >= 0 && j < muutettavat.size()) {
117             String s = muutettavat.get(j);
118             JLabel lbl = lblLista.get(h);
119             lbl.setText(s);
120             h--;
121             j++;
122         }
123     }

125     public void paivita(ArrayList<String> tekstit) {
126         int uusiKerros = liikkuminen.getKerros();
127         kartta = lg.getKartta();
128         viholliset = liikkuminen.getViholliset();
129         String kerros = "" + liikkuminen.getKerros();
130         f.invalidate();
131         for (int i = 0; i < sivunPituus; i++) {
132             for (int j = 0; j < sivunPituus; j++) {
133                 int index = (j * sivunPituus) + i;
134                 JPanel pan = list.get(index);
135                 pan.removeAll();
136                 if (i == 0 && j == 0) {
137                     pan.setBackground(Color.gray);
138                     Font font = new Font("Terminus", Font.PLAIN, 8);
139                     JLabel lbl = new JLabel("Floor");
140                     lbl.setFont(font);
141                     lbl.setBorder(BorderFactory.createEmptyBorder(-2 /*top*/, 0, 0, 0));
142                     JLabel krs = new JLabel(kerros);
143                     krs.setVerticalTextPosition(JLabel.BOTTOM);
144                     Font f = new Font("Terminus", Font.PLAIN, 10);
145                     krs.setFont(f);
146                     pan.add(lbl);
147                     pan.add(krs);
148                 } else if (kartta[j][i] == 0) {
149                     pan.setBackground(Color.black);
150                 } else {
151                     Font font = new Font("Terminus", Font.BOLD, 8);
152                     Font hpFont = new Font("Terminus", Font.PLAIN, 8);
153                     Pelaaja p = this.liikkuminen.getPelaaja();
154                     if (i == p.getKoordinaatit().getX() && j == p.getKoordinaatit().getY()) {
155                         JLabel lbl = new JLabel("@");
156                         lbl.setFont(font);
157                         lbl.setBorder(BorderFactory.createEmptyBorder(-2 /*top*/, 0, 0, 0));
158                         if (this.nykyinenKerros < uusiKerros) {
159                             pan.setBackground(Color.YELLOW);
160                             this.nykyinenKerros = uusiKerros;
161                         } else {
162                             pan.setBackground(Color.white);
163                         }
164                         JLabel hp = new JLabel(p.getHitPoints() + "/" + p.getHitPointsMax());
165                         hp.setFont(hpFont);
166                         hp.setVerticalTextPosition(JLabel.BOTTOM);
167                         pan.add(lbl);
168                         pan.add(hp);
169                     } else if (this.viholliset.tarkistaOtukset(i, j)) {
170                         pan.setBackground(Color.white);
171                         Otus o = this.viholliset.palautaPiirrettava(i, j);
172                         JLabel lbl = new JLabel(o.getTunnus());
173                         lbl.setFont(font);
174                         if (o.getTunnus().equals(" D ")) {
175                             lbl.setForeground(Color.RED);
176                             hpFont = new Font("Terminus", Font.PLAIN, 6);
177                         }
178                         lbl.setBorder(BorderFactory.createEmptyBorder(-2 /*top*/, 0, 0, 0));
179                         JLabel hp = new JLabel(o.getHitPoints() + "/" + o.getHitPointsMax());
180                         hp.setFont(hpFont);
181                         hp.setVerticalAlignment(JLabel.BOTTOM);
182                         pan.add(lbl);
183                         pan.add(hp);
184                     } else {
185                         if (kartta[j][i] == 5) {
186                             pan.setBackground(Color.white);
187                             JLabel lbl = new JLabel("%");
188                             pan.add(lbl);
189                         } else {
190                             pan.setBackground(Color.white);
191                         }
192                     }
193                 }
194             }
195         }
196         if (tekstit.size() > 0) {
197             MuutaLabelit(tekstit);
198         }
199         f.revalidate();
200         f.repaint();
201         this.kuoliko();
202         this.lvlUp();
203         this.pelinLoppu();
204     }

222     public void kuoliko() {
223         Object[] vaihtoEhdot = {
224             "Restart",
225             "Give up"
226         };
227         Pelaaja p = this.liikkuminen.getPelaaja();
228         if (p.getHitPoints() < 1) {
229             int reply = JOptionPane.showOptionDialog(f, "You have died! Do you want to restart?", "Death", JOptionPane.YES_NO_OPTION, JOptionPane.PLAIN_MESSAGE, null, vaihtoEhdot, vaihtoEhdot[0]);
230             if (reply == 0) {
231                 lg = new LuolaGeneraattori(sivunPituus);
232                 liikkuminen = new Liikkuminen(lg);
233                 viholliset = liikkuminen.getViholliset();
234                 kartta = lg.getKartta();
235                 k.setLiikkuminen(liikkuminen);
236                 this.tyhjennaLogi();
237                 this.lisaaLabelit();
238                 this.paivita(new ArrayList<String>());
239             } else {
240                 f.dispatchEvent(new WindowEvent(f, WindowEvent.WINDOW_CLOSING));
241             }
242         }
243     }

245     private void pelinLoppu() {
246         Object[] vaihtoEhdot = {
247             "Restart",
248             "Retire"
249         };
250         Pelaaja p = this.liikkuminen.getPelaaja();
251         if (this.liikkuminen.getKerros() == 50) {
252             f.invalidate();
253             for (int i = 0; i < sivunPituus; i++) {
254                 for (int j = 0; j < sivunPituus; j++) {
255                     int index = (j * sivunPituus) + i;
256                     JPanel pan = list.get(index);
257                     pan.removeAll();
258                     pan.setBackground(Color.WHITE);
259                 }
260             }
261             f.revalidate();
262             f.repaint();
263             int reply = JOptionPane.showOptionDialog(f, "You descended 50 floors and won the game! Do you want to restart or retire as a winner?", "Victory", JOptionPane.YES_NO_OPTION, JOptionPane.PLAIN_MESSAGE, null, vaihtoEhdot, vaihtoEhdot[0]);
264             if (reply == 0) {
265                 lg = new LuolaGeneraattori(sivunPituus);
266                 liikkuminen = new Liikkuminen(lg);
267                 viholliset = liikkuminen.getViholliset();
268                 kartta = lg.getKartta();
269                 k.setLiikkuminen(liikkuminen);
270                 this.tyhjennaLogi();
271                 this.lisaaLabelit();
272                 this.paivita(new ArrayList<String>());
273             } else {
274                 f.dispatchEvent(new WindowEvent(f, WindowEvent.WINDOW_CLOSING));
275             }
276         }

285     public void lvlUp() {
286         Pelaaja p = this.liikkuminen.getPelaaja();
287         if (p.getExperience() > 99) {
288             p.growExperience(-100);
289             String[] vaihtoEhdot = {
290                 "<html>Health + 5<br />(Delete or 1)</html>",
291                 "<html>Health Refill<br />(End or 2)</html>",
292                 "<html>Attack Power + 2<br />(Page Down or 3)</html>"
293             };
294             JDialog dia = new JDialog(f, "lvlup", true);
295             JPanel pan = new JPanel(new GridBagLayout());
296             JLabel uusi = new JLabel("Level up! You can choose from 3 options:");
297             Font font = new Font("Terminus", Font.PLAIN, 16);
298             uusi.setFont(font);
299             uusi.setHorizontalAlignment(JLabel.CENTER);
300             pan.setPreferredSize(new Dimension(400, 150));
301             dia.setResizable(false);
302             GridBagConstraints c = new GridBagConstraints();
303             c.gridx = 0;
304             c.gridy = 1;
305             c.weighty = 1;
306             c.gridwidth = 3;
307             pan.add(uusi, c);
308             JButton eka = new JButton();
309             eka.setText(vaihtoEhdot[0]);
310             eka.setFocusable(false);
311             eka.setEnabled(true);
312             eka.getInputMap(WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke(KeyEvent.VK_DELETE, 0), "hpUp");

314             eka.getActionMap().put("hpUp", new AbstractAction("hpUp") {
315                 @Override
316                 public void actionPerformed(ActionEvent e) {
317                     p.setHitPoints(p.getHitPoints() + 5);
318                     p.setHitPointsMax(p.getHitPointsMax() + 5);
319                     dia.dispose();
320                 }
321             });
322             eka.addActionListener(new ActionListener() {
323                 @Override
324                 public void actionPerformed(ActionEvent e) {
325                     p.setHitPoints(p.getHitPoints() + 5);
326                     p.setHitPointsMax(p.getHitPointsMax() + 5);
327                     dia.dispose();
328                 }
329             });
330             JButton toka = new JButton();
331             toka.setText(vaihtoEhdot[1]);
332             toka.setFocusable(false);
333             toka.setEnabled(true);
334             toka.getInputMap(WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke(KeyEvent.VK_END, 0), "hpFill");
335             toka.getInputMap(WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke(KeyEvent.VK_2, 0), "hpFill");
336             toka.getActionMap().put("hpFill", new AbstractAction("hpFill") {
337                 @Override
338                 public void actionPerformed(ActionEvent e) {
339                     p.setHitPoints(p.getHitPointsMax());
340                     dia.dispose();
341                 }
342             });
343             toka.addActionListener(new ActionListener() {
344                 @Override
345                 public void actionPerformed(ActionEvent e) {
346                     p.setHitPoints(p.getHitPointsMax());
347                     dia.dispose();
348                 }
349             });
350             JButton kolm = new JButton();
351             kolm.setText(vaihtoEhdot[2]);

353             kolm.setEnabled(true);

361                 }
362             });
363             kolm.addActionListener(new ActionListener() {
364                 @Override
365                 public void actionPerformed(ActionEvent e) {
366                     p.setAttackPower(p.getAttackPower() + 2);
367                     dia.dispose();
368                 }

370             for (int j = 0; j < 3; j += 2) {
371                 for (int i = 0; i < 3; i++) {
372                     GridBagConstraints co = new GridBagConstraints();
373                     co.gridx = i;
374                     co.gridy = j;
375                     co.weighty = 0;

381             co.gridx = 0;
382             co.gridy = 3;
383             co.weighty = 0.5;
